\begin{quotation}
	La tua vista ti può ingannare\ldots

	Il tuo udito ti può ingannare\ldots

	La tua esperienza ti può ingannare\ldots

	Anche la tua immaginazione ti può ingannare\ldots

	\emph{Ma la matematica no!}

	\hfill{---Throne of Magical Arcana}
\end{quotation}
